## Goal

把 v1-scope §3.3 Query 数据流落成可执行 LangGraph 工作流：
1. `TypedDict RagState` 定义 8 节点共享状态（query / history / rewrites / chunks / answer / sources / intent / error / thread_id / user_id / **trace_id**）—— r2-2026-06-12：M8 r2 新-1 揭示 M8 响应字段名与 M10 trace_id 命名断裂；M7 在 state 暴露 `trace_id`（`graph.ainvoke` 返回 state 含 trace_id），M8 透传到 HTTP 响应、M10 透传到 Langfuse 上下文字段，三者统一
2. 8 节点函数：`load_memory` → `classify` → `query_rewrite` → `retrieve` → `rerank` → `answer` → `save_memory_rag` / `save_memory_chitchat`（r2-2026-06-12 新-2 拆双节点） + chitchat 分支的 `answer_chitchat` 节点 = **9 个节点函数**（r1 标的 8 节点 = 7 主 + answer_chitchat 在 r2 拆 save_memory 后变 9）
3. 条件边：`classify` 节点判 `intent ∈ {retrieve, chitchat}` 决定后续走哪条分支；`answer → save_memory` 用常量边（V1 不需要路由）
4. `AsyncPostgresSaver` 注入到 `compile()`，thread_id 与 `chat_sessions.thread_id` 关联
5. SQLite 降级路径（PG 不可达 / `setup()` 权限不足时 fallback）
6. graph 调用方（M8）必须用 `asyncio.wait_for(graph.ainvoke(...), timeout=30)` 包装 invoke 防止 LLM 慢调用拖垮 API
7. 所有节点函数用 `safe_node` 装饰器包装，节点异常只设 `state.error` 不终止 graph
8. 端到端：`graph.invoke(query, thread_id)` 跑通，Postgres checkpoint tables 落数据，跨 invoke thread 复用历史

**不包含**（其他 M 负责）：M4-M6 ingest 写 chunk 到 OpenSearch（M7 只读）、M8 FastAPI 包装（含 `asyncio.wait_for` 超时）、M9 Gradio 集成、M10 业务级 Langfuse trace（M3 工厂已注入了 LLM callback，M10 加节点级 trace）

import logging
from app.config import settings
from langgraph.checkpoint.base import BaseCheckpointSaver

log = logging.getLogger(__name__)

async def make_checkpointer() -> BaseCheckpointSaver:
    """P0-3: 返回 BaseCheckpointSaver 实例，lifecycle 由调用方管理。
    P1-16: setup() 失败降级到 SQLite，不抛 PermissionError 阻塞 graph。"""
    if settings.checkpointer.dsn:
        try:
            from langgraph.checkpoint.postgres.aio import AsyncPostgresSaver
            cp = AsyncPostgresSaver.from_conn_string(
                settings.checkpointer.dsn,
                # P1-11: 显式连接池（langgraph-checkpoint-postgres 2.0.10 默认参数）
                # 默认 pool_size=5, max_overflow=10；M1 业务 DB pool_size=10+max_overflow=20
                # 总连接数 5+10+10+20 = 45 < PG max_connections=100
            )
            await cp.setup()
            log.info("checkpointer: postgres ready (pool=5+10)")
            return cp
        except (PermissionError, Exception) as e:
            log.warning("PG checkpointer setup failed (%s), falling back to SQLite", str(e))
    # SQLite 降级
    from langgraph.checkpoint.sqlite.aio import AsyncSqliteSaver
    cp = AsyncSqliteSaver.from_conn_string(settings.checkpointer.sqlite_path)
    await cp.setup()
    log.info("checkpointer: sqlite fallback at %s", settings.checkpointer.sqlite_path)
    return cp
```

from opensearchpy import AsyncOpenSearch
from app.config import settings

class Chunk(TypedDict):
    chunk_id: str
    doc_id: str
    text: str
    vector: list[float]
    metadata: dict
    _score: float | None          # OpenSearch kNN cosine 评分
    rerank_score: float | None    # P1-8: rerank 节点写入

async def retrieve(query: str, top_k: int = 10) -> list[Chunk]:
    # 1. embed query (M3 TEIEmbedder)
    from app.embedding.client import TEIEmbedder
    embedder = TEIEmbedder()
    [qvec] = await embedder.embed([query])

    # 2. kNN search — P1-9: index 名称从 settings 读，不硬编码
    # P1-13: V1.1 可在 query.knn.vector.filter 加 metadata filter
    client = AsyncOpenSearch(
        hosts=[settings.opensearch.url],
        connection_class=RequestsHttpConnection,
        pool_maxsize=settings.opensearch.pool_size,  # P2-6
        retry_on_timeout=True,
        max_retries=3,
    )
    resp = await client.search(
        index=settings.opensearch.index_name,
        body={
            "size": top_k,
            "query": {"knn": {"vector": {"vector": qvec, "k": top_k}}},

def dedupe_by_doc(chunks: list[Chunk]) -> list[Chunk]:
    """P1-10: 同 doc_id 保留最高 _score 的 chunk；首次出现位置优先。"""
    seen: dict[str, Chunk] = {}
    for c in chunks:
        doc_id = c["doc_id"]
        existing = seen.get(doc_id)
        if not existing or c.get("_score", 0) > existing.get("_score", 0):
            seen[doc_id] = c
    # 保持原顺序（首次出现位置优先），每个 doc_id 只出一次
    result = []
    emitted: set[str] = set()
    for c in chunks:
        doc_id = c["doc_id"]
        if doc_id not in emitted and seen.get(doc_id) is c:
            result.append(c)
            emitted.add(doc_id)
    return result
```

import json
import re

def _extract_json(text: str) -> str:
    """P1-5: 从 LLM 输出中提取 JSON 块（兼容 markdown 包裹 / 裸露 JSON / 字段名 scores 或 score）"""
    match = re.search(r"```(?:json)?\s*(\{.*?\})\s*```", text, re.DOTALL)
    if match:
        return match.group(1)
    match = re.search(r"\{.*?\}", text, re.DOTALL)
    if match:
        return match.group(0)
    return ""

def parse_json_scores(raw: str, n: int) -> list[float]:
    input_str = _extract_json(raw)
    if not input_str:
        return [5.0] * n
    try:
        data = json.loads(input_str)
        scores = data.get("scores", data.get("score", []))
        if not isinstance(scores, list) or len(scores) != n:
            return [5.0] * n
        return [max(0.0, min(10.0, float(s))) for s in scores]
    except (json.JSONDecodeError, ValueError, TypeError):
        return [5.0] * n
```

import logging
from functools import wraps
from app.graph.state import RagState

log = logging.getLogger(__name__)

class GraphNodeError(Exception):
    """节点级异常基类（包装后继续走 graph，不终止）"""
    node: str

class GraphNodeFallback(Exception):
    """节点降级标记（如 retrieve 不可用时降级到 fallback_answer）"""
    pass

def safe_node(node_name: str):
    """P0-8 装饰器：节点异常 → 设 state.error + 日志，不抛出。
    r2-2026-06-12 新-3 已修：asyncio.CancelledError（上游 wait_for 取消 / SIGTERM）
    与 KeyboardInterrupt（用户 Ctrl-C）必须放行，不吞掉 —— 否则
    M8 asyncio.wait_for 取消时 LangGraph 不会终止，shutdown 挂死；
    进程被 kill 时也走不到 handler。"""
    def decorator(func):
        @wraps(func)
        async def wrapper(state: RagState, *args, **kwargs) -> RagState:
            try:
                return await func(state, *args, **kwargs)
            except (asyncio.CancelledError, KeyboardInterrupt):
                # r2-2026-06-12 新-3：控制流异常原样上抛，**不** swallow 进 state.error
                raise
            except Exception as e:
                log.error("Node %s failed: %s", node_name, str(e), exc_info=True)
                return {
                    **state,
                    "error": f"{node_name}_failed: {str(e)[:200]}",
                }
        return wrapper
    return decorator

    g.add_node("save_memory_rag", save_memory_rag_node)
    g.add_node("save_memory_chitchat", save_memory_chitchat_node)

    g.add_edge(START, "load_memory")
    g.add_edge("load_memory", "classify")
    # r2-2026-06-12 新-1 已修：add_conditional_edges mapping 必须显式列出
    # route_after_classify 所有可能返回值；否则 class 是布尔判 True→retrieve / False→chitchat
    # （或 Literal）字面量未在 mapping 中时会抛 KeyError。
    # route_after_classify 显式 return "query_rewrite" / "answer_chitchat" / "END"
    # 三种（"END" 用于 classify 出现未预期 intent 时的安全 fallback）
    g.add_conditional_edges(
        "classify",
        route_after_classify,
        {
            "query_rewrite": "query_rewrite",
            "answer_chitchat": "answer_chitchat",
            "END": END,  # r2-2026-06-12 新-1：安全退出（不阻断用户、留 audit）
        },
    )
    g.add_edge("query_rewrite", "retrieve")
    g.add_edge("retrieve", "rerank")
    g.add_edge("rerank", "answer")
    # r2-2026-06-12 新-2：原 P0-5 常量边 "answer → save_memory" 改为条件边，
    # 路由函数 route_after_answer 按 state.intent 分流到 save_memory_rag
    # 或 save_memory_chitchat；P0-5 注已在新-2 注释中说明权衡。
    g.add_conditional_edges(
        "answer",
        route_after_answer,
        {
            "save_memory_rag": "save_memory_rag",
            "save_memory_chitchat": "save_memory_chitchat",
        },
    )
    # answer_chitchat 路径固定走 save_memory_chitchat
    g.add_edge("answer_chitchat", "save_memory_chitchat")
    g.add_edge("save_memory_rag", END)
    g.add_edge("save_memory_chitchat", END)

    return g.compile(
        checkpointer=checkpointer,
        interrupt_after=[],  # V1 无中断
        # P0-4: 节点数 8 < langgraph 1.0.5 默认 max_iterations=25, 安全
        # 单节点超时由 M8 调用方 asyncio.wait_for 包装
    )
```
